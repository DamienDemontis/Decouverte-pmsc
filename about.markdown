  document.addEventListener('DOMContentLoaded', function() {
    // Initialize Swiper
    var swiper = new Swiper(".specialtySwiper", {
      // Autoplay with 2s delay between movements instead of continuous
      autoplay: {
        delay: 2000, 
        disableOnInteraction: true, // Stop when user interacts
      },
      speed: 1000, // Faster transition speed for smoother experience
      loop: true,
      loopAdditionalSlides: 5, // Ensure smooth looping
      // Display 5 slides per view with centered effect
      slidesPerView: 5,
      spaceBetween: 20,
      centeredSlides: true,
      grabCursor: true,
      // Scale effect for centered slide
      effect: "coverflow",
      coverflowEffect: {
        rotate: 0,
        stretch: 0,
        depth: 100,
        modifier: 1,
        slideShadows: false,
      },
      // Navigation arrows
      navigation: {
        nextEl: '.swiper-button-next',
        prevEl: '.swiper-button-prev',
      },
      // Responsive breakpoints
      breakpoints: {
        // when window width is <= 1200px
        1200: {
          slidesPerView: 4,
          spaceBetween: 20
        },
        // when window width is <= 992px
        992: {
          slidesPerView: 3,
          spaceBetween: 15
        },
        // when window width is <= 768px
        768: {
          slidesPerView: 2,
          spaceBetween: 15,
          centeredSlides: false,
          effect: "slide"
        },
        // when window width is <= 480px
        480: {
          slidesPerView: 1,
          spaceBetween: 10,
          centeredSlides: false,
          effect: "slide"
        }
      }
    });
    
    // Improved interaction handling
    let isUserInteracting = false;
    
    // Animation on hover for cards within carousel
    const cards = document.querySelectorAll('.specialtySwiper .specialite-card');
    cards.forEach(card => {
      card.addEventListener('mouseenter', () => {
        // Pause autoplay on hover
        swiper.autoplay.stop();
      });
      card.addEventListener('mouseleave', () => {
        // Resume autoplay when mouse leaves only if user is not dragging
        if (!isUserInteracting) {
          swiper.autoplay.start();
        }
      });
    });
    
    // Handle manual interactions better
    swiper.el.addEventListener('mousedown', function() {
      isUserInteracting = true;
      swiper.autoplay.stop();
    });
    
    swiper.el.addEventListener('touchstart', function() {
      isUserInteracting = true;
      swiper.autoplay.stop();
    });
    
    // Resume autoplay after interaction ends and some delay
    document.addEventListener('mouseup', function() {
      if (isUserInteracting) {
        isUserInteracting = false;
        setTimeout(function() {
          swiper.autoplay.start();
        }, 1000); // 1s delay before resuming autoplay
      }
    });
    
    document.addEventListener('touchend', function() {
      if (isUserInteracting) {
        isUserInteracting = false;
        setTimeout(function() {
          swiper.autoplay.start();
        }, 1000); // 1s delay before resuming autoplay
      }
    });
  });
</script>
